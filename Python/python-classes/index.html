    Creating Objects with Super class

    Super Class:
    • An already defined class, that will be adopted by the current class.
    • Adds functionality of super class into current class.

      • Super Class 
      
class RobotIntro:
def __init__(self):
    self.name = ""

def intro(self):
    print(f"My name is Bond. {self.name} Bond!")
      Adding Super Class to current class 
      class RobotIntro:
    def __init__(self):
        self.name = ""

    def intro(self):
        print(f"My name is Bond. {self.name} Bond!")


class Robot4(RobotIntro):                   #Adding Super Class as an argument when building the class
    def __init__(self, name, weight):
        super().__init__()                  #Initializing the Super Class
        self.name = name
        self.weight = weight


r5 = Robot4("Kia", 87)
r5.intro()
      # Output:
My name is Bond. Kia Bond!

Recursion - Functions within functions

Solu: 
    def multiplication(multiplier):
    def myFunc(x):
        return multiplier * x

    return myFunc

multiplyBy3 = multiplication(3)
print(multiplyBy3(10))

def doAdditionON(x, operation):
    return operation(x)

result = doAdditionON(5, multiplyBy3)
print(result)
    # Output:
30
15
