Default Arguments

  • If no explicit values are passed, the default values will be used.
Default CSS
def sayHello(a, b="Portia"):
    print(f"Hello {a} & {b}")

sayHello("Tom", "Jerry")
sayHello("Ellen")
# Output:
Hello Tom & Jerry
Hello Ellen & Portia


*args

  • Allows a function to accept any number of arguments
Default CSS
def addNums (*num):
return sum(num)

print(addNums(1, 2, 3, 4, 5))
# Output:
15


*kwargs

  • Allows a function to accept any number of keyword arguments
Default CSS
def greet(*fnames):
print (f"Hello {fnames}")

greet("Jane", "July", "Tom")
# Output:
Hello ('Jane', 'July', 'Tom')
def greet(*fnames):
for i in range(len(fnames)):
print(f"Hello {fnames[i]}")

greet("Jane", "July", "Tom")
# Output:
Hello Jane
Hello July
Hello Tom
